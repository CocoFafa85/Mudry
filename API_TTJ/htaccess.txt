# Turn rewrite engine on
# Options + FollowSymlinks
Options +SymLinksIfOwnerMatch
RewriteEngine on

# map neat URL to internal URL
RewriteRule ^utilisateurs/list/$  apiUtilisateur.php?$method=GET [nc,qsa]
RewriteRule ^emplois/list/$  apiEmploi.php?$method=GET [nc,qsa]
RewriteRule ^postulants/list/$  apiPostulant.php?$method=GET [nc,qsa]

RewriteRule ^utilisateur/([0-9]+)/$   apiUtilisateur.php?$method=GET&id=$1 [nc,qsa]
RewriteRule ^connexion/([^/]+)/([^/]+)/$  apiPostulant.php?$method=GET&email=$1&mdp=$2 [nc,qsa]
RewriteRule ^emploi/([0-9]+)/$   apiEmploi.php?$method=GET&id=$1 [nc,qsa]
RewriteRule ^postuler/([0-9]+)/([0-1])/$  apiPostuler.php?$method=GET&id=$1&admis=$2 [nc,qsa]

RewriteRule ^ajoututilisateur/$   apiUtilisateur.php?$method=POST [nc,qsa]
RewriteRule ^ajoutpostulant/$   apiPostulant.php?$method=POST [nc,qsa]
RewriteRule ^ajoutemploi/$   apiEmploi.php?$method=POST [nc,qsa]
RewriteRule ^ajoutpostuler/$   apiPostuler.php?$method=POST [nc,qsa]

RewriteRule ^modificationutilisateur/$   apiUtilisateur.php?$method=PUT [nc,qsa]
RewriteRule ^modificationpostulant/$   apiPostulant.php?$method=PUT [nc,qsa]
RewriteRule ^modificationemploi/$   apiEmploi.php?$method=PUT [nc,qsa]

RewriteRule ^suppressionutilisateur/$   apiUtilisateur.php?$method=DELETE$id=$1 [nc,qsa]
RewriteRule ^suppressionpostulant/$   apiPostulant.php?$method=DELETE$id=$1 [nc,qsa]
RewriteRule ^suppressionemploi/$   apiEmploi.php?$method=DELETE$id=$1 [nc,qsa]
